-- Safe Loader
local function safeLoad(name, url)
    Rayfield:Notify({ Title = "Loading " .. name, Content = "Please wait...", Duration = 2 })
    local ok, err = pcall(function()
        loadstring(game:HttpGet(url))()
    end)
    if not ok then
        Rayfield:Notify({ Title = "Error Loading Script", Content = tostring(err), Duration = 5 })
    else
        Rayfield:Notify({ Title = "Loaded " .. name, Content = "Enjoy!", Duration = 2 })
    end
end

-- Update Checker
local function checkForUpdates()
    local ok, latest = pcall(function()
        return game:HttpGet(VERSION_URL)
    end)
    if ok and latest and latest ~= HUB_VERSION then
        Rayfield:Notify({
            Title = "Update Available!",
            Content = "Your version: " .. HUB_VERSION .. " | Latest: " .. latest,
            Duration = 5
        })
    else
        Rayfield:Notify({
            Title = "Up to Date",
            Content = "You're running the latest version!",
            Duration = 3
        })
    end
end

-- Search Tab
local SearchTab = Window:CreateTab("üîç Script Search")
SearchTab:CreateSection("Search for Scripts")
local ScriptLibrary = {
    ["Fly Script"] = "https://pastebin.com/raw/xyz123",
    ["ESP Script"] = "https://pastebin.com/raw/abc456",
    ["Speed Script"] = "https://pastebin.com/raw/def789"
}
SearchTab:CreateInput({
    Name = "Search Script",
    PlaceholderText = "Type script name...",
    RemoveTextAfterFocusLost = false,
    Callback = function(text)
        local found = false
        for name, url in pairs(ScriptLibrary) do
            if string.find(string.lower(name), string.lower(text)) then
                safeLoad(name, url)
                found = true
                break
            end
        end
        if not found then
            Rayfield:Notify({ Title = "No Script Found", Content = "Try another search term", Duration = 3 })
        end
    end
})
